def *
  raise StandardError.new("Not enough numbers on stack") if @stack.length<2
  num1 = pop
  num2 = pop
  result = num2 * num1
  push(result)
  return result
end

def round
  raise StandardError.new("Not enough numbers on stack") if @stack.length<1
  num = pop
  result = num.round
  push(result)
  return result
end
